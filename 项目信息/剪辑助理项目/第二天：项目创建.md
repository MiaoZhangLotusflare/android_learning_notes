#### 2.主界面的功能代码
```
package com.zhangmiao.clipassistant.Activity;

import android.content.Intent;
import android.support.v7.app.AppCompatActivity;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;

import com.zhangmiao.clipassistant.R;

/**
 * 主界面
 */
public class MainActivity extends AppCompatActivity {

    private static final String TAG = MainActivity.class.getSimpleName();

    private Button btn_localVideo;
    private Button btn_shootVideo;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        btn_localVideo = (Button) findViewById(R.id.activity_main_local_video_btn);
        btn_shootVideo = (Button) findViewById(R.id.activity_main_shoot_video_btn);
        btn_localVideo.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                gotoChooseLocalVideo();
            }
        });
        btn_shootVideo.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                gotoShootVideo();
            }
        });
    }

    private void gotoChooseLocalVideo() {
        Intent intent = new Intent(this, LocalVideoActivity.class);
        startActivity(intent);
    }

    private void gotoShootVideo() {
        Intent intent = new Intent(this, ShootVideoActivity.class);
        startActivity(intent);
    }
}

```
